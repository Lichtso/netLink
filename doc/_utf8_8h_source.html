147 template<typename iterator>
148 size_t byteSize(iterator i) {
149 size_t s;
150 char32_t c;
151 start:
152 c = i[0];
153 if((c&0x80) == 0)
154 return 1;
155 s = countLeadingOneBits(c<<24);
156 if(s == 1) {
157 decrement(i);
158 goto start;
159 } else
160 return (s > 4) ? -1 : s;
161 }

164 template<typename iterator>
165 char32_t toUtf32(iterator i) {
166 char32_t c;
167 start:
168 c = i[0];
169 if((c&0x80) == 0)
170 return c;
171 switch(countLeadingOneBits(c << 24)) {
172 case 1:
173 decrement(i);
174 goto start;
175 case 2:
176 return (c&0x1F)<<6 | (i[1]&0x3F);
177 case 3:
178 return (c&0x0F)<<12 | (i[1]&0x3F)<<6 | (i[2]&0x3F);
179 case 4:
180 return (c&0x0F)<<18 | (i[1]&0x3F)<<12 | (i[2]&0x3F)<<6 | (i[3]&0x3F);
181 default:
182 return -1;
183 }
184 }
