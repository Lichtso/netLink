161  template<typename iterator>
162  char32_t toUtf32(iterator i) {
163  char32_t c;
164  start:
165  c = i[0];
166  if((c&0x80) == 0)
167  return c;
168  switch(countLeadingOneBits(c << 24)) {
169  case 1:
170  decrement(i);
171  goto start;
172  case 2:
173  return (c&0x1F)<<6 | (i[1]&0x3F);
174  case 3:
175  return (c&0x0F)<<12 | (i[1]&0x3F)<<6 | (i[2]&0x3F);
176  case 4:
177  return (c&0x0F)<<18 | (i[1]&0x3F)<<12 | (i[2]&0x3F)<<6 | (i[3]&0x3F);
178  default:
179  return -1;
180  }
